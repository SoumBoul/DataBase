create procedure AjouterDepartement
(
@DapartementID int,
@NomDepartement nvarchar(50),
@Responsable nvarchar(50)
)

as
begin

insert into DepartementsRH(DepartmentID,NomDepartement,Responsable)
values(@DapartementID,@NomDepartement,@Responsable);

end

CREATE PROCEDURE AfficherDepartements
AS
BEGIN
SELECT DepartmentID, NomDepartement, Responsable
FROM DepartementsRH
ORDER BY NomDepartement;
END

create procedure MettreAJourResponsable

@DepartementID int,

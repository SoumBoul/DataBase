select *, CAST(NumberOfVehicules as float ) /CAST(TotalOfVehicules as float) from
 (
 select Makes.Make,
 count(*)NumberOfVehicules, (select count(*) from  VehicleDetails) as TotalOfVehicules

 from VehicleDetails
 inner join Makes on VehicleDetails.MakeID= Makes.MakeID
 where Year between 1950 and 2000
 group by Make 
 )R1
 order by NumberOfVehicules desc

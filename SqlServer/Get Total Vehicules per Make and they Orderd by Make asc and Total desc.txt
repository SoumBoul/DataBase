select  distinct Makes.Make,DriveTypes.DriveTypeName, count(*) Vehicules
 from DriveTypes inner join VehicleDetails on VehicleDetails.DriveTypeID= DriveTypes.DriveTypeID
 inner join Makes on Makes.MakeID= VehicleDetails.MakeID

 group by Make,DriveTypes.DriveTypeName 
 order by Make asc , Vehicules desc
